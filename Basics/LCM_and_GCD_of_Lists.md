# LCM of a list of numbers
def lcm_list(b):
    def find_lcm(num1, num2):
        if(num1>num2):
            num = num1
            den = num2
        else:
            num = num2
            den = num1
        rem = num % den
        while(rem != 0):
            num = den
            den = rem
            rem = num % den
        gcd = den
        lcm = int(int(num1 * num2)/int(gcd))
        return lcm
    
    num1 = b[0]
    num2 = b[1]
    
    lcm = find_lcm(num1, num2)
    
    for i in range(2, len(b)):
        lcm = find_lcm(lcm, b[i])
    return lcm
# GCD of list of numbers
def gcd_list(l):
    def find_gcd(x, y):
         
        while(y):
            x, y = y, x % y
         
        return x
    
    num1 = l[0]
    num2 = l[1]
    gcd = find_gcd(num1, num2)
    
    for i in range(2, len(l)):
        gcd = find_gcd(gcd, l[i])
    return gcd
